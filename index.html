// BotFeature record
{
  featureType: "pending_approvals",
  menuItem: {
    labelTokens: [ /* expression */ ],
    // default: "⏳ אישורים ({count})"
  },
  itemTemplate: {
    titleTokens: [ /* expression */ ],
    // default: "{employee.firstName} — {event.name}"
  },
  requiredPermissions: ["approve_payroll"],
  config: {
    requireRejectionReason: true,
    allowFieldEditing: true
  }
}
